//all lessons stored as an array of objects
const lessons=[
    {
        title:"HTML Basics",
        description:"Learn how to structure a webpage using HTML5",
        image:"assets/images/html-course.png"
    },
    {
        title:" CSS Flexbox",
        description:"Master modern layout with Flexbox",
        image:"assets/images/html-course.png"
    },
    {
        title:" JavaScript Essentials",
        description:"Understand JS from basics to DOM",
        image:"assets/images/html-course.png"
    },
    
];

//target the lessons container
const lessoncontainer=document.getElementById("lesson-container");

//render each lessson card dynamically
lessons.forEach(lesson=>{
    const lessoncard=document.createElement("section");
    lessoncard.className="lesson-card";

  lessoncard.innerHTML=`
  <img src="${lesson.image}" alt="${lesson.title}">
  <h2>${lesson.title}</h2>
  <p>${lesson.description}</p>
  <button onclick="startLesson('${lesson.title}')">Start Lesson</button>

  `;

  lessoncontainer.append(lessoncard);

});

//funvtion to handle lesson start
function startLesson(title){
    alert(`Lesson Started:${title}`);
    localStorage.setItem("lesson_starded",title);
    console.log("lesson saved:",title);
}
